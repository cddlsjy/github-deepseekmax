Okay, converting a full-fledged Python Tkinter application to an Android Kotlin application using Jetpack Compose is a substantial task, as it involves a complete rewrite due to fundamental differences in language, UI framework, threading model, and platform APIs.

I will provide a structured Android project with the core functionalities translated to Kotlin and Jetpack Compose. This includes:

Project Structure: Standard Android project with Activities, ViewModels, and Composables.
Language: Python to Kotlin.
UI Framework: Tkinter to Jetpack Compose.
Networking: requests to Retrofit + OkHttp + Kotlin Coroutines.
Configuration: json file to Android DataStore (preferred over SharedPreferences for modern apps).
Threading: Python's implicit threading to explicit Kotlin Coroutines for background tasks.
File System: os to Android's storage APIs (Context.filesDir, ContentResolver for downloads).
Messages/Dialogs: Tkinter messagebox to Android AlertDialog / Snackbar.
Font Scaling: Custom CompositionLocalProvider in Compose.
Treeview: This is the most complex component to port directly. I'll implement a simplified, expandable list using LazyColumn for the file tree.
File Download: Uses Android's Storage Access Framework via ActivityResultLauncher to let the user pick a save location.

class ConfigManager(private val context: Context) {

    companion object {
        private val TOKEN_KEY = stringPreferencesKey("github_token")
        private val FONT_SCALE_KEY = doublePreferencesKey("font_scale")
        private val LAST_SEARCH_QUERY_KEY = stringPreferencesKey("last_search_query")
        private val LAST_REPO_FULL_NAME_KEY = stringPreferencesKey("last_repo_full_name")
    }

    suspend fun loadConfig(): AppConfig {
        return context.dataStore.data.map { preferences ->
            AppConfig(
                token = preferences[TOKEN_KEY] ?: "",
                fontScale = (preferences[FONT_SCALE_KEY] ?: Constants.DEFAULT_FONT_SCALE.toDouble()).toFloat(),
                lastSearchQuery = preferences[LAST_SEARCH_QUERY_KEY] ?: "",
                lastRepoFullName = preferences[LAST_REPO_FULL_NAME_KEY] ?: ""
            )
        }.first()
    }

    suspend fun saveConfig(config: AppConfig) {
        context.dataStore.edit { preferences ->
            preferences[TOKEN_KEY] = config.token
            preferences[FONT_SCALE_KEY] = config.fontScale.toDouble()
            preferences[LAST_SEARCH_QUERY_KEY] = config.lastSearchQuery
            preferences[LAST_REPO_FULL_NAME_KEY] = config.lastRepoFullName
        }
    }

    suspend fun saveToken(token: String) {
        context.dataStore.edit { preferences ->
            preferences[TOKEN_KEY] = token
        }
    }

    suspend fun saveFontScale(scale: Float) {
        context.dataStore.edit { preferences ->
            preferences[FONT_SCALE_KEY] = scale.toDouble()
        }
    }

    suspend fun saveLastSearchQuery(query: String) {
        context.dataStore.edit { preferences ->
            preferences[LAST_SEARCH_QUERY_KEY] = query
        }
    }

    suspend fun saveLastRepoFullName(fullName: String) {
        context.dataStore.edit { preferences ->
            preferences[LAST_REPO_FULL_NAME_KEY] = fullName
        }
    }
}

// --- Network Client Setup ---

class AuthInterceptor(private val token: String) : Interceptor {
    override fun intercept(chain: Interceptor.Chain): Response {
        val original = chain.request()
        val requestBuilder = original.newBuilder()
            .header("Authorization", "token $token")
            .header("Accept", "application/vnd.github.v3+json")
        val request = requestBuilder.build()
        return chain.proceed(request)
    }
}

object GitHubHttpClient {
    private var retrofit: Retrofit? = null
    private var okHttpClient: OkHttpClient? = null

    @Synchronized
    fun getClient(token: String): GitHubApi {
        if (retrofit == null || okHttpClient?.interceptors?.find { it is AuthInterceptor && it.token != token } != null) {
            val logging = HttpLoggingInterceptor().apply {
                level = HttpLoggingInterceptor.Level.BODY // Adjust level as needed
            }

            okHttpClient = OkHttpClient.Builder()
                .addInterceptor(AuthInterceptor(token))
                .addInterceptor(logging)
                .build()

            retrofit = Retrofit.Builder()
                .baseUrl(Constants.GITHUB_BASE_URL)
                .client(okHttpClient!!)
                .addConverterFactory(GsonConverterFactory.create())
                .build()
        }
        return retrofit!!.create(GitHubApi::class.java)
    }

    // For direct OkHttp calls (e.g., artifact download redirects)
    @Synchronized
    fun getPlainClient(token: String? = null): OkHttpClient {
        val builder = OkHttpClient.Builder()
        if (token != null) {
            builder.addInterceptor(AuthInterceptor(token))
        }
        return builder.build()
    }

    // Helper to URL-encode paths for contents API
    fun encodePath(path: String): String {
        return URLEncoder.encode(path, StandardCharsets.UTF_8.toString())
    }
}

package com.example.githubexplorer.data.github

import android.util.Base64
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.async
import kotlinx.coroutines.awaitAll
import kotlinx.coroutines.withContext
import retrofit2.HttpException
import java.io.File
import java.io.IOException

class GitHubRepository(private val api: GitHubApi, private val okHttpClient: OkHttpClient) {

    suspend fun getUser(): GitHubUser = api.getUser()

    suspend fun listUserRepos(): List<GitHubRepo> = withContext(Dispatchers.IO) {
        val allRepos = mutableListOf<GitHubRepo>()
        var page = 1
        while (true) {
            val repos = api.listUserRepos(page = page)
            if (repos.isEmpty()) break
            allRepos.addAll(repos)
            if (repos.size < 100) break
            page++
        }
        allRepos
    }

    suspend fun searchRepos(query: String): List<GitHubRepo> = withContext(Dispatchers.IO) {
        val allRepos = mutableListOf<GitHubRepo>()
        var page = 1
        while (true) {
            val response = api.searchRepos(query = query, page = page)
            val repos = response.items
            if (repos.isEmpty()) break
            allRepos.addAll(repos)
            if (repos.size < 100) break
            page++
        }
        allRepos
    }

    suspend fun getRepoDefaultBranch(owner: String, repo: String): String {
        return api.getRepo(owner, repo).defaultBranch
    }

    suspend fun getRepoTree(owner: String, repo: String, branch: String, recursive: Boolean = true): List<TreeEntry> = withContext(Dispatchers.IO) {
        val ref = api.getBranchRef(owner, repo, branch)
        val commitSha = ref.`object`.sha
        val recursiveParam = if (recursive) 1 else null
        api.getRepoTree(owner, repo, commitSha, recursiveParam).tree
    }

    @Throws(Exception::class)
    suspend fun getFileContent(owner: String, repo: String, path: String, branch: String = "main"): Any? = withContext(Dispatchers.IO) {
        try {
            val encodedPath = GitHubHttpClient.encodePath(path)
            val response = api.getFileContent(owner, repo, encodedPath, branch)
            if (response.type == "dir") {
                return@withContext null // It's a directory, not a file content
            }

            if (response.encoding == "base64") {
                val decoded = Base64.decode(response.content, Base64.DEFAULT)
                try {
                    return@withContext decoded.toString(Charsets.UTF_8)
                } catch (e: Exception) {
                    // Not UTF-8, return bytes
                    return@withContext decoded
                }
            }
            return@withContext response.content
        } catch (e: HttpException) {
            if (e.code() == 404) { // File not found or not a file
                return@withContext null
            }
            throw e
        } catch (e: Exception) {
            throw e
        }
    }

    // --- Actions API ---

    suspend fun listWorkflows(owner: String, repo: String): List<Workflow> {
        return api.listWorkflows(owner, repo).workflows
    }

    suspend fun dispatchWorkflow(owner: String, repo: String, workflowId: Long, ref: String) {
        api.dispatchWorkflow(owner, repo, workflowId, WorkflowDispatchPayload(ref))
    }

    suspend fun listWorkflowRuns(owner: String, repo: String, workflowId: Long, status: String? = null, perPage: Int = 30, page: Int = 1): List<WorkflowRun> {
        return api.listWorkflowRuns(owner, repo, workflowId, status, perPage, page).workflowRuns
    }

    suspend fun getRunJobs(owner: String, repo: String, runId: Long): List<Job> {
        return api.getRunJobs(owner, repo, runId).jobs
    }

    suspend fun getJobLog(owner: String, repo: String, jobId: Long): String = withContext(Dispatchers.IO) {
        // GitHub API for job logs can return a redirect. Retrofit handles redirects for JSON,
        // but for plain text body it might need manual handling if the log is very large.
        // Retrofit's default client usually follows redirects for text/plain as well.
        // However, the original Python code had a specific check, so keeping an eye on it.
        api.getJobLog(owner, repo, jobId)
    }

    suspend fun getRunLogsCombined(owner: String, repo: String, runId: Long): String = withContext(Dispatchers.IO) {
        val jobs = getRunJobs(owner, repo, runId)
        val logParts = jobs.map { job ->
            async {
                try {
                    val log = getJobLog(owner, repo, job.id)
                    "--- Job: ${job.name} (${job.id}) ---\n$log\n"
                } catch (e: Exception) {
                    "--- Job: ${job.name} (${job.id}) - Error loading log ---\n${e.message}\n"
                }
            }
        }.awaitAll()
        logParts.joinToString("\n")
    }


    suspend fun listRunArtifacts(owner: String, repo: String, runId: Long): List```html  {
        return api.listRunArtifacts(owner, repo, runId).artifacts
    }

    suspend fun downloadArtifact(artifactUrl: String, destinationFile: File) = withContext(Dispatchers.IO) {
        okHttpClient.newCall(okhttp3.Request.Builder().url(artifactUrl).build()).execute().use { response ->
            if (!response.isSuccessful) {
                throw IOException("Failed to download artifact: ${response.code} ${response.message}")
            }
            response.body?.byteStream()?.use { input ->
                destinationFile.outputStream().use { output ->
                    input.copyTo(output)
                }
            } ?: throw IOException("Empty response body for artifact download")
        }
    }
}

    private fun buildTree(flatEntries: List<TreeEntry>): List<TreeEntryNode> {
        val rootNodes = mutableListOf<TreeEntryNode>()
        val allNodesMap = mutableMapOf<String, TreeEntryNode>()

        // Create all nodes and map them by path
        for (entry in flatEntries) {
            val node = TreeEntryNode(
                path = entry.path,
                name = entry.path.substringAfterLast('/'),
                type = entry.type,
                children = mutableListOf()
            )
            allNodesMap[entry.path] = node
        }

        // Build the hierarchy
        for (node in allNodesMap.values) {
            val parentPath = node.path.substringBeforeLast('/', "")
            if (parentPath.isEmpty()) {
                rootNodes.add(node)
            } else {
                allNodesMap[parentPath]?.children?.add(node)
            }
        }

        // Sort children alphabetically, directories first
        fun sortChildren(nodes: List<TreeEntryNode>) {
            nodes.forEach { node ->
                node.children.sortWith(compareBy({ it.type != "tree" }, { it.name.lowercase(Locale.ROOT) }))
                sortChildren(node.children)
            }
        }
        sortChildren(rootNodes)
        rootNodes.sortWith(compareBy({ it.type != "tree" }, { it.name.lowercase(Locale.ROOT) }))

        return rootNodes
    }

@Composable
fun TreeItem(node: TreeEntryNode, viewModel: MainViewModel, expandedNodesMap: Map<String, Boolean>, depth: Int) {
    val isExpanded = expandedNodesMap[node.path] ?: false
    val horizontalPadding = (depth * 16).dp

    Row(
        modifier = Modifier
            .fillMaxWidth()
            .clickable {
                if (node.type == "tree") {
                    viewModel.toggleNodeExpansion(node.path)
                } else {
                    viewModel.displayFileContent(node.path)
                }
            }
            .padding(start = horizontalPadding, vertical = 4.dp, end = 8.dp),
        verticalAlignment = Alignment.CenterVertically
    ) {
        if (node.type == "tree") {
            Icon(
                imageVector = if (isExpanded) Icons.Filled.KeyboardArrowDown else Icons.Filled.KeyboardArrowRight,
                contentDescription = if (isExpanded) "Collapse" else "Expand",
                modifier = Modifier.size(20.dp)
            )
        } else {
            Spacer(modifier = Modifier.width(20.dp)) // Align files with expanded folders
        }
        Icon(
            imageVector = if (node.type == "tree") Icons.Filled.Folder else Icons.Filled.Description,
            contentDescription = node.type,
            modifier = Modifier.size(20.dp).padding(horizontal = 4.dp)
        )
        Text(node.name, maxLines = 1, overflow = TextOverflow.Ellipsis)
    }

    if (isExpanded && node.type == "tree") {
        node.children.forEach { child ->
            TreeItem(node = child, viewModel = viewModel, expandedNodesMap = expandedNodesMap, depth = depth + 1)
        }
    }
}

class MainActivity : ComponentActivity() {

    private lateinit var configManager: ConfigManager

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        configManager = ConfigManager(this)

        setContent {
            GitHubExplorerTheme {
                val coroutineScope = rememberCoroutineScope()
                var loggedInUser by remember { mutableStateOf<GitHubUser?>(null) }
                var gitHubRepository by remember { mutableStateOf<GitHubRepository?>(null) }
                var initialConfig by remember { mutableStateOf<AppConfig?>(null) }
                var showLogin by remember { mutableStateOf(true) }

                // Load initial config and try auto-login on startup
                LaunchedEffect(Unit) {
                    val config = configManager.loadConfig()
                    initialConfig = config
                    val token = config.token
                    if (token.isNotBlank()) {
                        try {
                            val api = GitHubHttpClient.getClient(token)
                            val repository = GitHubRepository(api, GitHubHttpClient.getPlainClient(token))
                            val user = repository.getUser()
                            gitHubRepository = repository
                            loggedInUser = user
                            showLogin = false // Auto-login successful
                        } catch (e: Exception) {
                            // Auto-login failed, show login screen
                            showLogin = true
                            // Optionally show a toast about auto-login failure
                        }
                    }
                }

                if (showLogin) {
                    val loginViewModel: LoginViewModel = viewModel(
                        factory = LoginViewModelFactory(LocalContext.current)
                    )
                    LoginScreen(
                        viewModel = loginViewModel,
                        onLoginSuccess = { repo, user ->
                            gitHubRepository = repo
                            loggedInUser = user
                            showLogin = false
                            coroutineScope.launch {
                                // Reload config after successful login to ensure it's up to date
                                // for MainViewModel initialization
                                initialConfig = configManager.loadConfig()
                            }
                        }
                    )
                } else if (gitHubRepository != null && loggedInUser != null && initialConfig != null) {
                    val mainViewModel: MainViewModel = viewModel(
                        factory = MainViewModelFactory(
                            LocalContext.current,
                            gitHubRepository!!,
                            initialConfig!!
                        )
                    )
                    MainScreen(viewModel = mainViewModel)
                }
            }
        }
    }

    // ViewModel Factories
    class LoginViewModelFactory(private val context: android.content.Context) : ViewModelProvider.Factory {
        override fun <T : ViewModel> create(modelClass: Class<T>): T {
            if (modelClass.isAssignableFrom(LoginViewModel::class.java)) {
                @Suppress("UNCHECKED_CAST")

Explanation and Differences:

Entry Point: Android apps start with an Activity. MainActivity.kt handles the initial login flow.
UI: Jetpack Compose replaces Tkinter widgets with @Composable functions. Layouts are handled declaratively using Column, Row, Modifier.weight(), etc.
State Management: mutableStateOf, mutableStateListOf, MutableStateFlow are used in ViewModels to manage UI state. Changes to these states automatically trigger UI recomposition.
Asynchronous Operations: All network requests and disk I/O are performed using Kotlin Coroutines (viewModelScope.launch, withContext(Dispatchers.IO)), ensuring the UI thread remains responsive.
API Client: requests is replaced by Retrofit (for structured API calls) and OkHttp (for direct binary downloads like artifacts). AuthInterceptor handles adding the token to every request.
Configuration: ConfigManager uses Android DataStore to persist the token and font scale across app launches, similar to the .github_browser_config.json file.
Login Flow: The LoginScreen uses a LoginViewModel to handle token input and validation. On successful login, MainActivity switches to MainScreen.
Main Application Structure:
MainScreen uses Scaffold, TopAppBar, BottomAppBar for basic app structure.
HorizontalPager with TabRow creates the tabbed interface (Repositories, Directory Tree, Actions).
LazyColumn is used for efficient display of long lists (repositories, tree items, workflow runs).
Treeview (DirectoryTreeTab): A custom recursive @Composable TreeItem is used to render the directory tree with indentation and expandable/collapsible folders, mimicking the Tkinter Treeview functionality.
File Content: ScrolledText is replaced by a Text composable within a verticalScroll modifier.
Actions Tab: ExposedDropdownMenuBox replaces ttk.Combobox for workflow selection and status filtering. LazyColumn displays workflow runs.
Artifact Download: Uses Android's ActivityResultContracts.CreateDocument to prompt the user to choose a save location, then OkHttp performs the stream download.
Error Handling & Feedback: errorMessage in LoginViewModel and showSnackbar, showDialog in MainViewModel are used to display AlertDialog and Snackbar messages, similar to Tkinter's messagebox.
Font Scaling: CompositionLocalProvider is used in MainScreen to override the default MaterialTheme.typography with scaled versions based on the fontScale setting, applying it to all child composables. Restarting the activity (currentContext.recreate()) is a common way to apply theme changes in Android, even in Compose, if the theme is deeply integrated.
